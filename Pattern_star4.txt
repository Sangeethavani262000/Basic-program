import java.util.Scanner;
class Pattern1{
public static void main(string[] args){
Scanner sc= new Scanner(System.in);
int n=sc.nextInt();
int i,j,k;
  for(i=1;i<=n;i++){
     for(j=n-1;j>=i;j--){
	System.out.print("");
	}
    for(k=1;k<=i;k++){
        System.out.print("*");
      }
     System.out.println();
   }
 }
}
